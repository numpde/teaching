    \subsection{}
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    
    Fix $m > 0$ and $g > 0$.
    %
    Consider 
    $
        E_\text{I}(x, v) := \tfrac12 m v^2 + g x
    $
    for
    $(x, v) \in \IR^2$.
    %
    \begin{itemize}
    \item 
        In the $x$-$v$ plane,
        sketch the level curves of $E_\text{I}$.
    \item
        If $E_\text{I}(x(t), v(t))$ is constant
        and $\dot{x} = v$, what is $\dot{v}$?
    \item
        Sketch $x$ and $v$ as functions of $t$.
    \item
        Compute $F = m \ddot{x}$.
    \end{itemize}
